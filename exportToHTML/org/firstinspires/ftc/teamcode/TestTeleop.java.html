10   public class  TestTeleop extends LinearOpMode {
11   
12       //Declares Variables to prevent NAN
13       double shooterSetpoint;
14       double intakePower;
15       double stagerPower;
16       double ring1Sensor;
17       double ring2Sensor;
18       double ring3Sensor;
19       double stopper = .3;
20       double xSpeedSetPoint = 1;
21       double highestMotorPower;
22       double LFM;
23       double speed;
24       double LBM;
25       double RFM;
26       double RBM;
27       boolean stagerLoop = false;
28       boolean stagerControl = false;
29       boolean shooterControl = false;
30       boolean gPadBControl = false;
31       double leftG1StickPoint;
32       double shooterVelocity;
33       double shooterActualVelocity;
34       double shooterPorportional;
35       double shooterPM;
36       double shooterDM;
37       double shooterLastError;
38       double shooterDerivitive;
39       double shooterCorrection;
40       double shooterError;
41       double timepassed;
42       double wobbleSet;
43       double wobbleCurrent;
44       double wobbleError;
45       double wobblePower;
46       double wobbleP = 1.1;
47       double GRIP_S = .6;
48       double shooterLastEncoder;
49       double SOTCurrent;
50       double SOTSet = 1.47;
51       double lastTime;
52       double SOTError;
53       double SOTPower;
54       double SOTP = -20;
55       double shooterFSM = 0;
56       double WB_FSM = 5;
57       boolean WBControl = false;
58       @Override
59       public void runOpMode() {
60           //Calling upon the HardwareMap
61           RobotHardware robot = new RobotHardware();
62           robot.init(hardwareMap);
63           double wobbleEndSet = robot.WB_PT.getVersion();
64           //Waits for the play button to be pressed
65           waitForStart();
66           //main loop
67           while (opModeIsActive()) {
68               //all sensor readings
69               //Takes a reading from all of the distance sensors
70               ring1Sensor = robot.Ring1_DS.getDistance(DistanceUnit.INCH);
71               ring2Sensor = robot.Ring2_DS.getDistance(DistanceUnit.INCH);
72               ring3Sensor = robot.Ring3_DS.getDistance(DistanceUnit.INCH);
73               //Takes potentiometer reading and writes it to a variable for use later in program
74               SOTCurrent = robot.SOT_PT.getVoltage();
75               wobbleCurrent = robot.WB_PT.getVoltage();
76               //Using a finite State Machine to easily control the stager and intake.
77               //Also using custom one button on/off system where we use a boolean to see if the state has changed and then do an action
78               if(gamepad1.a && stagerControl == false){
79                   if(intakePower == 0){
80                       shooterFSM = 1;
81                   }else{
82                       shooterFSM = 2;
83                   }
84                   stagerControl = true;
85               }else if(!gamepad1.a){
86                   stagerControl = false;
87               }
88               //stage 0 shuts all motors off and this stager is starting stage
89               if(shooterFSM == 0){
90                   intakePower = 0;
91                   stagerPower = 0;
92               }
93               //stage 1 is intaking stage. Sets ring stopper to closed and intakes until sensors see that there is 3 rings in robot.
94               if(shooterFSM == 1){
95                   stopper = .3;
96                   if(ring1Sensor < 2 && ring2Sensor < 2 && ring3Sensor < 4){
97                       intakePower = 0;
98                       stagerPower = 0;
99                       shooterFSM = 2;
100                  }else {
101                      intakePower = -1;
102                      stagerPower = -1;
103                  }
104              }
105              //stage 2 is shooting stage. If button b is pressed rings shoot other wise motors are off
106              if(shooterFSM == 2) {
107                  if (gamepad1.b) {
108                      stopper = .5;
109                      stagerPower = -1;
110                      intakePower = 0;
111                  } else {
112                      stagerPower = 0;
113                      intakePower = 0;
114                  }
115              }
116              //Lets us reverse the intake direction if intake gets jammed
117              if(gamepad1.back){
118                  intakePower = 1;
119              }
120              //Shooter Control
121              //Manual adjusting the setpoint to adjust last second if needed
122              if(gamepad1.y){
123                  SOTSet = SOTSet - .003;
124              }else if(gamepad1.x){
125                  SOTSet = SOTSet + .003;
126              }
127              //Shooter Angle PID Loop follo the setpoint set above
128              SOTError = SOTSet - SOTCurrent;
129              SOTPower = SOTError * SOTP;
130              //Flywheel speed setpoint control. We use our custom one button on/off system to use the left bumper to set the shooter speed.
131              if (gamepad1.left_bumper && shooterControl == false) {
132                  if(shooterSetpoint == 0){
133                      shooterSetpoint = 1900;//set point is 1900 encoder ticks per second
134  
135                  }else{
136                      shooterSetpoint = 0;
137                      shooterCorrection = 0;//we set both of these variables to ensure that neither one has power
138                  }
139                  shooterControl = true;
140              }else if(!gamepad1.left_bumper){
141                  shooterControl = false;
142              }
143              //This is thhe Flywheels PID. This makes sure the Shotoer speed is the same no matter the battery power
144              if(shooterSetpoint !=0){
145                  shooterPM = 15;//set the proportional multiplier for the shooter speed
146                  timepassed = getRuntime() - lastTime;
147                  shooterActualVelocity = Math.abs(robot.SOT_M.getCurrentPosition()-shooterLastEncoder)/timepassed;
148                  lastTime = getRuntime();
149                  shooterLastEncoder = robot.SOT_M.getCurrentPosition();
150                  shooterError = shooterSetpoint - shooterActualVelocity;
151                  shooterPorportional = shooterError *  shooterPM;
152                  shooterCorrection = shooterPorportional;
153              }
154              //finite State machine for the wobble goal.
155              //We use the custom one button cycle to switch between each state.
156              if(gamepad1.left_trigger > .05 && WBControl == false){
157                  if(WB_FSM < 4){
158                      WB_FSM = WB_FSM + 1;
159                  }else{
160                      WB_FSM = 0;
161                  }
162                  WBControl = true;
163              }else if(gamepad1.left_trigger < .05){
164                  WBControl = false;
165              }
166              //State 0 is claw open and in grabbing position
167              if(WB_FSM == 0) {
168                  wobbleEndSet = 2.324;
169                  GRIP_S = .1;
170              }else if(WB_FSM == 1){//state 1 is grab wobble goal while still in down position
171                  GRIP_S = .7;
172              }else if(WB_FSM == 2){//Brings wobble goal arm in stored position to drive to wall
173                  wobbleEndSet = .6;
174              }else if(WB_FSM == 3) {//Brings wobble goal above wall, gripper still closed
175                  wobbleEndSet = 1;
176              }else if(WB_FSM == 4){// Gripper opens
177                  GRIP_S = .1;
178              }else if(WB_FSM == 5){//Closes Claw
179                  GRIP_S = .7;
180              }
181              //Lets up manually change set point if needed. Set EndSet to current set point + or - t o allow the arm to go to its current position + or - the setpoint.
182              if (gamepad1.dpad_up) {
183                  wobbleEndSet = wobbleSet - .1;
184              } else if (gamepad1.dpad_down) {
185                  wobbleEndSet = wobbleSet + .1;
186              }
187              //This slowly brings the arm to the endsetpoint to ensure little wear from arm bannging on the wheels
188              if(wobbleEndSet > .2 + wobbleSet){
189                  wobbleSet = wobbleSet + .05;
190              }else if(wobbleEndSet < .2 - wobbleSet){
191                  wobbleSet = wobbleSet - .05;
192              }else{
193                  wobbleSet = wobbleEndSet;
194              }
195              //PID to control the wobble arm to go to desired set point
196              wobbleError = wobbleSet - robot.WB_PT.getVoltage();
197              wobblePower = wobbleError * wobbleP;
198              //Drivetrain Control
199              //Sets slow speed by using the right bumper
200              if(gamepad1.right_bumper){
201                  xSpeedSetPoint = .5;
202              }else{
203                  xSpeedSetPoint = 1;
204              }
205              //sets calculation variables to let us calculate drivetrain direction
206              double x = -gamepad1.left_stick_x;
207              double y = -gamepad1.left_stick_y;
208              double z = -gamepad1.right_stick_x;
209              //calculates motor speed and direction
210              LFM = y - (x + z);
211              LBM = y + (x - z);
212              RFM = y + (x + z);
213              RBM = y - (x - z);
214              //We use highest motor power to make sure no wheel speed ever goes over 1 and lets su strafe more accuratly
215              highestMotorPower = Math.max(Math.max(Math.abs(LFM), Math.abs(LBM)), Math.max(Math.abs(RFM), Math.abs(RBM)));
216              leftG1StickPoint = Math.sqrt((gamepad1.left_stick_x*gamepad1.left_stick_x) + (gamepad1.left_stick_y*gamepad1.left_stick_y));
217              //Since we use highest motor power to make sure no motor power goes over 1, there will always be a motor going full power
218              //So we use the joysticks again to make apply a ratio and the set motor power
219              if (Math.abs(gamepad1.right_stick_x) <.01){
220                  speed = leftG1StickPoint;
221              }else if(Math.abs(gamepad1.left_stick_x + gamepad1.left_stick_y) < .02){
222                  speed = Math.abs(gamepad1.right_stick_x);
223              } else{
224                  speed = (leftG1StickPoint + Math.abs(gamepad1.right_stick_x))/2;
225              }
226              //Displaying Telemetry
227              telemetry.addData("WB_FSM", WB_FSM);
228              telemetry.addData("WBSET", wobbleSet);
229              telemetry.addData("WBerror", wobbleError);
230              telemetry.addData("WB_PT", robot.WB_PT.getVoltage());
231              telemetry.addData("E1", robot.LF_M.getCurrentPosition() * 0.00436111);
232              telemetry.addData("E2", robot.LB_M.getCurrentPosition() * 0.00436111);
233              telemetry.addData("E3", robot.RF_M.getCurrentPosition() * 0.00436111);
234              telemetry.addData("ShooterEncoder", robot.SOT_M.getCurrentPosition());
235              telemetry.addData("ShooterPower", (shooterSetpoint+shooterCorrection)/2800);
236              telemetry.addData("shooterError", shooterError);
237              telemetry.addData("Shooter Correction", shooterCorrection);
238              telemetry.addData("shooterSetpoint", shooterSetpoint);
239              telemetry.addData("LeftX", gamepad1.left_stick_x);
240              telemetry.addData("LeftY", gamepad1.left_stick_y);
241              telemetry.addData("speed", speed);
242              telemetry.addData("LFM", robot.LF_M.getPower());
243              telemetry.addData("shooterSetpoint", shooterSetpoint);
244              telemetry.addData("stagerControl", stagerControl);
245              telemetry.addData("DS1", ring1Sensor);
246              telemetry.addData("DS2", ring2Sensor);
247              telemetry.addData("DS3", ring3Sensor);
248              telemetry.addData("Potentiometer", robot.SOT_PT.getVoltage());
249              telemetry.addData("SOTPower", SOTPower);
250              telemetry.addData("SOTCurrent", SOTCurrent);
251              telemetry.addData("SOTError", SOTError);
252              telemetry.addData("SOTSet", SOTSet);
253              telemetry.addData("WobblePower", wobblePower);
254              telemetry.addData("WBmotor", robot.WB_M.getPower());
255              telemetry.update();
256  
257              //Setting Motor Power
258              robot.LF_M.setPower(((LFM/highestMotorPower) * speed)* xSpeedSetPoint);
259              robot.LB_M.setPower(((LBM/highestMotorPower) * speed)* xSpeedSetPoint);
260              robot.RF_M.setPower(((RFM/highestMotorPower) * speed)* xSpeedSetPoint);
261              robot.RB_M.setPower(((RBM/highestMotorPower) * speed)* xSpeedSetPoint);
262              robot.WB_M.setPower(wobblePower);
263              robot.SOT_M.setPower((shooterSetpoint + shooterCorrection)/2800);
264              robot.IN_M.setPower(intakePower);
265              robot.STG_M.setPower(-stagerPower);
266              robot.GRIP_S.setPosition(GRIP_S);
267              robot.SOT_S.setPower(SOTPower);
268              robot.STOP_S.setPosition(stopper);
269          }
270      }
