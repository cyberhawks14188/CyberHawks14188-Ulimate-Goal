3    public class Odometry {
4        double deltaE1, deltaE2, deltaE3;
5        double e1Previous, e2Previous, e3Previous;
6        final double encoderWheelDistance = 14.42206;
7        final double COUNTS_PER_INCH = 236.5;
8        final double encoderWheelDistanceInch = encoderWheelDistance * COUNTS_PER_INCH;
9        double thetaChange, thetaInRadians;
10       double e2WithOffSet;
11       final double e2XOffSet = 3000;
12       double yAverage;
13       double yCoordinatePosition, xCoordinatePosition;
14       double e1Current, e2Current, e3Current;
15   
16       public void OdometryCalc(double e1current, double e2current, double e3current){
17           //finds the change in the encoders from the last loop cycle
18           e1Current = e1current;
19           e2Current = e2current;
20           e3Current = e3current;
21           deltaE1 = e1Current - e1Previous;
22           deltaE2 = e2Current - e2Previous;
23           deltaE3 = e3Current - e3Previous;
24           //calculates the angle in radians
25           thetaChange = (deltaE1 - deltaE3)/ encoderWheelDistanceInch;
26           thetaInRadians = thetaInRadians + thetaChange;
27           //calculates the e2 reading without the angle values affecting in it
28           e2WithOffSet = deltaE2 - (thetaChange * e2XOffSet);
29           //finds the average between the forward facing encoders to find the center of the robot
30           yAverage = (deltaE1 + deltaE3)/2;
31   
32           yCoordinatePosition = yCoordinatePosition + (yAverage*Math.sin(thetaInRadians)) + (e2WithOffSet*Math.cos(thetaInRadians));
33           xCoordinatePosition = xCoordinatePosition + (yAverage*Math.cos(thetaInRadians)) - (e2WithOffSet*Math.sin(thetaInRadians));
34           //sets varibles to the current encoder reading to set Delta variables in the next loop cycle
35           e1Previous = e1Current;
36           e2Previous = e2Current;
37           e3Previous = e3Current;
38       }
39       //returns the values to use in other classes
40       public double odoXReturn(){return xCoordinatePosition/COUNTS_PER_INCH;}
41       public double odoYReturn(){return yCoordinatePosition/COUNTS_PER_INCH;}
42       public double thetaINRadiansReturn(){return thetaInRadians;}
43       public double thetaInDegreesReturn(){return Math.toDegrees(thetaInRadians) % 360;}
44   }
