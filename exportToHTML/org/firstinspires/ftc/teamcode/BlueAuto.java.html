595      public void Movement(double X_EndSetpoint, double Y_EndSetpoint, double Z_setpoint, double Slow_Down_Distance, double accelerationDistance) {
596          //Sets our PID Multipliers
597          X_PM = .15;
598          X_IM = .000000001;
599          X_DM = .35;
600          Y_PM = .2;
601          Y_IM = .000000001;
602          Y_DM = .2;
603          Z_PM = .45;
604          Z_IM = .000000001;
605          Z_DM = .55;
606          VPM = .7;
607          VIM = .00000001;
608          VDM = .9;
609          Slope_X_DM = .4;
610          Slope_Y_PM = .4;
611          Slope_X_PM = .4;
612          Slope_Y_DM = .4;
613          minimumAccelerationVelocity = 12;
614          minimumVelocity = 2;
615          //If we only want our robot to turn we only run the Z PID by setting all other multipliers to 0
616          if(JustTurn == 1){
617              X_PM = 0;
618              X_IM = 0;
619              X_DM = 0;
620              Y_PM = 0;
621              Y_IM = 0;
622              Y_DM = 0;
623              Slope_X_DM = 0;
624              Slope_Y_PM = 0;
625              Slope_X_PM = 0;
626              Slope_Y_DM = 0;
627              Z_PM = .15;
628              Z_DM = .2;
629  
630          }
631  
632          //Gets encoder Positions
633          //Converts our encoder ticks to inches
634          E1 = robot.LF_M.getCurrentPosition() * 0.00436111;
635          E2 = -robot.LB_M.getCurrentPosition() * 0.00436111;
636          E3 = robot.RF_M.getCurrentPosition() * 0.00436111;
637          //Sets encoders to .001 at begining to prevent null error
638          if (E1 == 0) {
639              E1 = .001;
640          }
641          if (E2 == 0) {
642              E2 = .001;
643          }
644  
645          if (E3 == 0) {
646              E3 = .001;
647          }
648          //XYZ PID
649          //Gets our robot moving in the direction of the endsetpoints
650  
651  
652          //Finds the diffrence of E1 and E3
653          z_encoder_diffrence = E1 - E3;
654          //Finds the difference between our setpoint and our current position
655          z_error = Z_setpoint - z_encoder_diffrence;
656          //Z Porportional
657          //Narrows down the rate of change by multiplying the error by a proportional multiplier
658          z_porportional = Z_PM * z_error;
659          // Z Intergral
660          //Finds the Sum of the Errors
661          Z_Sum_of_Errors = Z_Sum_of_Errors + z_error;
662          Z_Intergral = Z_Sum_of_Errors * Z_IM;
663          //Z Derivitive
664          //Finds the difference between the current error and last loop's error
665          Z_Diffrence_of_Errors = z_error - Z_Last_Error;
666          Z_Derivitive = Z_Diffrence_of_Errors * Z_DM;
667          Z_Last_Error = z_error;
668          z = z_porportional + Z_Intergral + Z_Derivitive;
669          //Finds the difference between our setpoint and our current position
670          x_error = X_EndSetpoint - E2;
671          //X Porportional
672          //Narrows down the rate of change by multiplying the error by a proportional multiplier
673          x_porportional = x_error * X_PM;
674          //X Intergral
675          //Finds the Sum of the Errors
676          X_Sum_of_Errors = X_Sum_of_Errors + x_error;
677          X_Intergral = X_Sum_of_Errors * X_IM;
678          //X Derivitve
679          //Finds the difference between the current error and last loop's error
680          X_Diffrence_of_Errors = x_error - X_Last_Error;
681          X_Derivitive = X_Diffrence_of_Errors * X_DM;
682          X_Last_Error = x_error;
683  
684          //Finds Average Of E1 & E3
685          Y_Average = (E1 + E3) / 2;
686          //finds the average of E1 and E3
687          //Finds the difference between our setpoint and our current position
688          y_error = Y_EndSetpoint - Y_Average;
689          //Y Porportional
690          //Narrows down the rate of change by multiplying the error by a proportional multiplier
691          y_porportional = Y_PM * y_error;
692          // Y Intergral
693          //Finds the Sum of the Errors
694          Y_Sum_of_Errors = Y_Sum_of_Errors + y_error;
695          Y_Intergral = Y_Sum_of_Errors * Y_IM;
696          //Y Derivitive
697          //Finds the difference between the current error and last loop's error
698          Y_Diffrence_of_Errors = y_error - Y_Last_Error;
699          Y_Derivitive = Y_Diffrence_of_Errors * Y_DM;
700          Y_Last_Error = y_error;
701  
702  
703          //Uses pythagorean therom to find how far we are from the end setpoints compared to our current position on the field our starting position
704          Distance = Math.sqrt((Math.pow((Y_EndSetpoint-Last_Y_EndSetpoint), 2)) + (Math.pow((X_EndSetpoint-Last_X_EndSetpoint), 2)));
705          //Calculates our Y_A2 by finding the difference between where we want to go and where we currently are
706          Y_A2 = Y_EndSetpoint - Y_Average;
707          //Calculates our X_B2 by finding the difference between where we want to go and where we currently are
708          X_B2 = X_EndSetpoint - E2;
709          //Calculates Distance from our end point from our current position
710          Distance_From = Math.sqrt(Math.pow(Y_A2, 2) + (Math.pow(X_B2, 2)));
711          //A if that only runs once, used to not allow our code to break out of the loop immediately
712          if (breakout == 1) {
713              lastDistanceFrom = Distance_From;
714              breakout = 0;
715          }
716  
717          //Gets our current run time
718          time = getRuntime();
719          //Finds the time passed from this loop cycle to the last loop cycle
720          time_passed = time - (previousTime);
721          //Velocity is the number of inches travled in a second
722          actualVelocity = Math.abs((Distance_From - lastDistanceFrom)) / time_passed;
723          //Sets previous time to time
724          previousTime = time;
725          //Sets lastDistanceFrom to our current Distance_From
726          lastDistanceFrom = Distance_From;
727          //Acceleration Code
728          //Will run when our distance that we will accelerate for is greater than our our distance traveled
729          if (accelerationDistance >= Distance - Distance_From) {
730          //Creates a line using slope that our velocity setpoint
731          //The farther we are from our starting position the more
732              velocitySetpoint = ((Distance - Distance_From) * (targetVelocity / accelerationDistance));
733              if(velocitySetpoint <= minimumAccelerationVelocity){
734                  velocitySetpoint = minimumAccelerationVelocity;
735              }
736          }
737          //Run the deacceleration when our distance from is less then the Slow_Down_Distance
738          if (Distance_From <= Slow_Down_Distance) {
739              //Will ramp down from our target veloicty to 0 based on our distance
740              velocitySetpoint = Distance_From * (targetVelocity / Slow_Down_Distance) + minimumVelocity;
741          }
742          //Allows us to call to a varible outside of the method
743          storingVarible1 = Y_EndSetpoint;
744          storingVarible2 = X_EndSetpoint;
745          //Velocity limits
746          if (velocitySetpoint <= minimumVelocity) {
747              velocitySetpoint = minimumVelocity;
748          }
749          //Velcotiy Max speed per second in inches
750          if (velocitySetpoint >= 50) {
751              velocitySetpoint = 50;
752          }
753          //If our robot has stopped moving and we want to be moving then incress the velocity setpoint to get us to move
754          if(actualVelocity == 0 && velocitySetpoint >.1){
755              velocitySetpoint = velocitySetpoint + 4;
756          }
757          //Runs and PID on the velocity of our robot
758          //Allows us to maintain the speed we want
759  
760          //Find the error between desired velcotiy and the current robot velocity
761          velocityError = velocitySetpoint - actualVelocity;
762          //Velocity Porportional
763          velocityPorportion = velocityError * VPM;
764          //Velocity Intergral
765          velocitySumOfErrors = velocitySumOfErrors + velocityError;
766          velocityIntergral = velocitySumOfErrors * VIM;
767          //Velocity Derivitive
768          velocityDiffrenceOfErrors = velocityError - velocityLastError;
769          velocityLastError = velocityError;
770          veloictyDerivitive = velocityDiffrenceOfErrors * VDM;
771          //Finds the correction our robot has to do to get back to the desired setpoint
772          velocityCorrection = (velocityPorportion + velocityIntergral + veloictyDerivitive);
773          //Find our robot's slope based on our starting position to our end position
774          slope = ((Y_EndSetpoint - Last_Y_EndSetpoint) / (X_EndSetpoint - Last_X_EndSetpoint));
775          
776          //Equate our maximum speed by the slope that we are going
777          //Since when going sideways our robot can only go 30in/s and 50in/s when going forward
778          //If our rise is higher than our run
779          if(Math.abs(slope) >= 1) {
780              maximumVelocity = 40 + (10-((1/Math.abs(slope))*10));
781          }
782          //If our run is higher then our rise
783          else if(Math.abs(slope) < 1){
784              maximumVelocity = 40 - (10-(Math.abs(slope) * 10));
785          }
786          //Finds our Y intercept
787          Y_intercept = Y_EndSetpoint-(X_EndSetpoint*slope);
788          if (Y_EndSetpoint - Last_Y_EndSetpoint != 0 && X_EndSetpoint - Last_X_EndSetpoint != 0) {
789              Y_Setpoint = slope * E2+Y_intercept;
790              X_Setpoint = (Y_Average-Y_intercept) / slope;
791              telemetry.addData("1", "1");
792          }
793          //Horizonatal line
794          //Have to do this since a horizontal lines slope is 0
795          else if (Y_EndSetpoint - Last_Y_EndSetpoint == 0) {
796              maximumVelocity = 30;
797              Y_Setpoint = Y_EndSetpoint;
798              X_Setpoint = E2;
799              telemetry.addData("2", "2");
800          }
801          //Vertical Line
802          //Have to do this since a vertical lines slope is undefined
803          else if (X_EndSetpoint - Last_X_EndSetpoint == 0) {
804              maximumVelocity = 50;
805              Y_Setpoint = Y_Average;
806              X_Setpoint = X_EndSetpoint;
807              telemetry.addData("3", "3");
808          }
809          //When we are within 4 inches of target hold at the target instead of following the line
810          if (Distance_From <= 4) {
811  
812              Y_Setpoint = Y_EndSetpoint;
813              X_Setpoint = X_EndSetpoint;
814  
815          }
816          //Find error between the desired point on line and our currenet E2 reading
817          Slope_Y_error = Y_Setpoint - Y_Average;
818          //Finds error between the desired point on line and our curernt E1 and E3 average
819          Slope_X_Error = X_Setpoint - E2;
820          //Runs the Y Slope PD
821          Slope_Y_porportional = Slope_Y_error * Slope_Y_PM;
822          Slope_Y_Derivitive = (Slope_Y_error - Slope_Y_Last_error) * Slope_Y_DM;
823          Slope_Y_Last_error = Slope_Y_error;
824          //Find the sum of the porportional and the derivitive
825          Slope_Y_Correction = Slope_Y_porportional + Slope_Y_Derivitive;
826          //Runs the X SLope PD
827          Slope_X_Porportional = Slope_X_Error * Slope_X_PM;
828          Slope_X_Derivitive = (Slope_X_Error - Slope_X_last_error) * Slope_X_DM;
829          Slope_X_last_error = Slope_X_Error;
830          //Find the sum of the  porprotional and the dervititive
831          Slope_X_Correction = Slope_X_Porportional + Slope_X_Derivitive;
832  
833          //Equates our y direction correction
834          y = y_porportional + Y_Intergral + Y_Derivitive + Slope_Y_Correction;
835          //Equates our x direction correction
836          x = x_porportional + X_Intergral + X_Derivitive + Slope_X_Correction;
837          //Runs our motor equation method
838          MotorEquation();
839          //Runs our telementry method
840          telemetry();
841      }

843      public void MotorEquation() {
844          //Motor equation from the slope and XYZ pid
845          LF_Direction = y+(x+z);
846          LB_Direction = y-(x-z);
847          RF_Direction = y-(x+z);
848          RB_Direction = y+(x-z);
849          //Finds Highest power out of the drive motors
850          //We do this to be able to then get a -1-1 reading on our motors
851          Highest_Motor_Power = Math.max(Math.max(Math.abs(RF_Direction), Math.abs(RB_Direction)), Math.max(Math.abs(LF_Direction), Math.abs(LB_Direction)));
852          //Sets motors
853          //Determines our speed of our motors by the velocity correction and setpoint
854          robot.LF_M.setPower(((velocityCorrection+velocitySetpoint)/maximumVelocity) * (LF_Direction/Highest_Motor_Power));
855          robot.LB_M.setPower(((velocityCorrection+velocitySetpoint)/maximumVelocity) * (LB_Direction/Highest_Motor_Power));
856          robot.RF_M.setPower(((velocityCorrection+velocitySetpoint)/maximumVelocity) * (RF_Direction/Highest_Motor_Power));
857          robot.RB_M.setPower(((velocityCorrection+velocitySetpoint)/maximumVelocity) * (RB_Direction/Highest_Motor_Power));
858      }
